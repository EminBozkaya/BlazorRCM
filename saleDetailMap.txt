using RCMServerData.Models;

namespace RCMServerData.EFMappingToSQL
{
    public class SaleDetailMap
    {
        public void Configure(EntityTypeBuilder<SaleDetail> builder)
        {
            builder.HasKey(x => x.Id)
                .HasName("pk_SaleDetail_id");

            builder.Property(x => x.Id)
                .HasColumnName("Id")
                .HasColumnType("uuid")
                .HasDefaultValueSql("UUID_GENERATE_V4()")
                .IsRequired();

            builder.Property(x => x.SId)
                .HasColumnName("PId")
                .HasColumnType("bigint")
                .IsRequired();

            builder.Property(x => x.PId)
                .HasColumnName("PId")
                .HasColumnType("smallint")
                .IsRequired();

            builder.Property(x => x.Price)
                    .HasColumnName("Price")
                    .HasColumnType("decimal")
                    .HasPrecision(10, 4)
                    .IsRequired();

            builder.Property(x => x.Qty)
                    .HasColumnName("Qty")
                    .HasColumnType("smallint")
                    .IsRequired();

            builder.Property(x => x.Portion)
                    .HasColumnName("Portion")
                    .HasColumnType("decimal")
                    .IsRequired();

            builder.Property(x => x.Total)
                    .HasColumnName("Total")
                    .HasColumnType("decimal")
                    .HasPrecision(10, 4)
                    .IsRequired();

            builder.Property(x => x.Note)
                    .HasColumnName("Note")
                    .HasColumnType("nvarchar")
                    .HasMaxLength(100);

            builder.Property(x => x.CutOff)
                    .HasColumnName("CutOff")
                    .HasColumnType("decimal")
                    .HasPrecision(10, 4);

            builder.Property(x => x.IsActive)
                    .HasColumnName("IsActive")
                    .HasColumnType("boolean")
                    .IsRequired();

            builder.Property(x => x.ModifiedBy)
                    .HasColumnName("ModifiedBy")
                    .HasColumnType("int");

            builder.Property(x => x.ModifiedTime)
                    .HasColumnName("ModifiedTime")
                    .HasColumnType("timestamp");




            builder
                .HasOne(e => e.Sale)
                .WithMany(e => e.SaleDetails)
                .HasForeignKey(f => f.SId);

            //builder
            //    .OwnsMany(x=>x.Products)
            //    .OwnsOne(p => p.SaleDetails);

        }
    }
}
